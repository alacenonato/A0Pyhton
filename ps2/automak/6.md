## 🔧 Script corrigido

Execute os comandos abaixo:

### 1. Remova o script antigo e crie um novo:

```bash
cd "/media/alace/My Passport/2POPS"
rm -f criar_elfs.sh
```

### 2. Crie o novo script corrigido:

```bash
cat > criar_elfs.sh << 'EOF'
#!/bin/bash

echo "========================================"
echo "  Criando arquivos ELF para jogos VCD"
echo "========================================"
echo ""

CONTADOR=0
CRIADOS=0

for vcd in *.VCD; do
    # Verifica se existe pelo menos um arquivo VCD
    if [ ! -f "$vcd" ]; then
        echo "Nenhum arquivo .VCD encontrado!"
        break
    fi
    
    # Remove a extensão .VCD para pegar o nome base
    nome_base="${vcd%.VCD}"
    
    # Nome do ELF com prefixo SB.
    elf_name="SB.${nome_base}.ELF"
    
    # Verifica se o ELF já existe
    if [ -f "$elf_name" ]; then
        echo "[EXISTE] $elf_name"
    else
        # Copia o POPSTARTER.ELF para criar o novo ELF
        cp POPSTARTER.ELF "$elf_name"
        echo "[CRIADO] $elf_name"
        CRIADOS=$((CRIADOS + 1))
    fi
    CONTADOR=$((CONTADOR + 1))
done

echo ""
echo "========================================"
echo "Total de jogos encontrados: $CONTADOR"
echo "Arquivos ELF criados: $CRIADOS"
echo "========================================"
EOF
```

### 3. Execute o script:

```bash
bash criar_elfs.sh
```
